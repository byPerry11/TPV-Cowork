-- Create Notifications Table
CREATE TABLE IF NOT EXISTS notifications (
    id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
    user_id UUID REFERENCES auth.users(id) NOT NULL,
    type TEXT NOT NULL CHECK (type IN ('friend_request', 'project_invite', 'checkpoint_completed', 'project_completed')),
    title TEXT NOT NULL,
    message TEXT NOT NULL,
    reference_id UUID, -- Can point to project_id, friend_request_id, etc.
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- RLS Policies
ALTER TABLE notifications ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can view their own notifications"
    ON notifications FOR SELECT
    USING (auth.uid() = user_id);

CREATE POLICY "Users can update their own notifications (mark as read)"
    ON notifications FOR UPDATE
    USING (auth.uid() = user_id);

CREATE POLICY "Users can insert notifications for others (e.g. sending invites)"
    ON notifications FOR INSERT
    WITH CHECK (true); -- Allow any authenticated user to send a notification (controlled by app logic)
